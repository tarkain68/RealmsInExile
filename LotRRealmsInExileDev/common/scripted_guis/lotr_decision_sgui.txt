lotr_request_prospecting_contract_as_landed = {
	effect = {
		clear_variable_list = potential_prospecting_county
		remove_variable = selected_county

		every_held_county = {
			limit = {
				any_county_province = {
					OR = {
						terrain = mountains
						terrain = halls
						terrain = desert_mountains
					}
				}
				NOT = { any_county_province = { valid_mining_activity_location = yes } }
			}
			save_scope_as = prospecting_county
			root = {
				add_to_variable_list = {
					name = potential_prospecting_county
					target = scope:prospecting_county
				}
			}
		}
	}
}

lotr_request_prospecting_clear_variables = {
	effect = {
		clear_variable_list = potential_prospecting_county
		remove_variable = selected_county
	}
}

lotr_request_prospecting_select_county = {
	saved_scopes = {
		selected_county
	}
	effect = {
		if = {
			limit = {
				has_variable = selected_county
				var:selected_county = scope:selected_county
			}
			remove_variable = selected_county
		} else = {
			set_variable = {
				name = selected_county
				value = scope:selected_county
			}
		}
	}
}

lotr_bountig_build_list_of_independent_easterling_lords = {
	effect = {
		clear_variable_list = independent_easterlings
		title:e_rhovanion = {
			ordered_de_jure_county_holder = {
				limit = {
					is_independent_ruler = yes
					OR = {
						culture = culture:igath
						culture = culture:asdriag
						culture = culture:odhriag
						culture = culture:magriag
					}
				}
				order_by = max_military_strength
				check_range_bounds = no
				min = 0
				save_scope_as = easterling_ruler
				root = {
					add_to_variable_list = {
						name = temporary_list
						target = scope:easterling_ruler
					}
				}
			}
		}
		title:e_rhun = {
			ordered_de_jure_county_holder = {
				limit = {
					is_independent_ruler = yes
					OR = {
						culture = culture:igath
						culture = culture:asdriag
						culture = culture:odhriag
						culture = culture:magriag
					}
				}
				order_by = max_military_strength
				check_range_bounds = no
				min = 0
				save_scope_as = easterling_ruler
				root = {
					add_to_variable_list = {
						name = temporary_list
						target = scope:easterling_ruler
					}
				}
			}
		}

		ordered_in_list = {
			variable = temporary_list
			order_by = max_military_strength
			check_range_bounds = no
			min = 0
			save_scope_as = easterling_ruler
			root = {
				add_to_variable_list = {
					name = independent_easterlings
					target = scope:easterling_ruler
				}
			}
		}
		clear_variable_list = temporary_list
	}
}
